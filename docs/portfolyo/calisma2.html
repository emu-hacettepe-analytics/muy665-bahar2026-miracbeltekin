6. NA Değerlerini Değiştirme

new_data <- polls_us_election_2016

for (col in names(new_data)) {
  if (is.numeric(new_data[[col]])) {
    new_data[[col]][is.na(new_data[[col]])] <- my_birth_year + k
  } else if (is.character(new_data[[col]]) || is.factor(new_data[[col]])) {
    if (is.factor(new_data[[col]])) {
      new_data[[col]] <- as.character(new_data[[col]])
      new_data[[col]][is.na(new_data[[col]])] <- paste0(my_first, "_", k)
      new_data[[col]] <- as.factor(new_data[[col]])
    } else {
      new_data[[col]][is.na(new_data[[col]])] <- paste0(my_first, "_", k)
    }
  }
}
